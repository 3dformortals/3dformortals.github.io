• Map interface kotlin.collections
    • Map<K, V> is collection that holds pairs of objects (keys and values). weblink
    MutableMap<K, V> is modifiable collection that holds pairs of objects (keys and values). weblink var map = mapOf( "abc" to "textkey", 4 to "numberkey", '3' to "charkey" ) //{a=textkey, 4=numberkey, 3=charkey}
    println( map[0] ) //null
    println( map["abc"] ) //textkey
    println( map.size ) //3
    println ( map.keys ) //[abc, 4, 3]
    println ( map.values ) //[textkey, numberkey, charkey]
    println ( map.entries ) //[abc=textkey, 4=numberkey, 3=charkey]

    mapID = mapID.plus( Pair( 1, 10.0 ) ) //{1=10.0}
    mapID = mapID.plus( 2 to 20.0 ) //{1=10.0, 2=20.0}
    // mapID += Pair( 3 to 30.0 ) //error
    mapID = mapID.plus( 2 to 40.0 ) //{1=10.0, 2=40.0}
    mapID = mapID.minus( 2 ) //{1=10.0}
    println( mapID.get( 1 ) ) //10.0

    var mMap = mutableMapOf( "def" to "textkey", 4.0 to "numberkey", 'x' to "charkey" )
    mMap.clear() //{ }
    mMap.putAll( map ) //{abc=textkey, 4=numberkey, 3=charkey}
    mMap.putAll( mutableMapOf( "3" to "stringkey" ) ) //{abc=textkey, 4=numberkey, 3=charkey, 3=stringkey}
    mMap.remove( '3' ) //{abc=textkey, 4=numberkey, 3=stringkey}
    mMap.remove( "3", "value" ) //{abc=textkey, 4=numberkey, 3=stringkey}
    println( mMap.get( '3' ) ) //null
    println( mMap.get( "3" ) ) //stringkey
    mMap.set( "3", "value" ) //{abc=textkey, 4=numberkey, 3=value}
    mMap.put("3", "text3") //{abc=textkey, 4=numberkey, 3=text3}
    mMap.putAll( mapOf( Pair("3", "string3"), '4' to "char4" ) ) //{abc=textkey, 4=numberkey, 3=string3, 4=char4}

    /*add to mutable map element from other mutable map */
    val map:MutableMap = mutableMapOf()
    val map1 = mutableMapOf("test1" to "value1")
    val map2 = mutableMapOf("test2" to "value2", "test3" to "value3")
    val map3 = mutableMapOf("test4" to "value4", "test5" to "value5")
    //map.put("test1", map1["test1"]) //error
    map.put("test1", map1["test1"]!!) //correct dirty, can throw exception, if map1 have no "test1" key
    map.put("test1", map1["test1"] ?: "") //correct with default empty string "" as value, if map1 have no "test1" key
    for ((key,value) in map2) map.put("new_"+key, value) //correct and clean, no need use ?: or !!
    for ((key,value) in map3) map["new_"+key] = value) //correct and clean, no need use ?: or !!
