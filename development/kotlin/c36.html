• Return multivalues from function
    • That comfort return two values from function you can use data class declaration and using. weblink data class Result(val dummyInt: Int, val dummyDouble: Double)

    fun x2x3dummy(dummy:Int): Result { return Result( dummy*2, ( dummy*3 ).toDouble() ) }

    var (x2, x3) = x2x3dummy(3) //destructuring declarations support
    print("int_result = $x2 double_result = $x3") //int_result = 6 double_result = 9.0
