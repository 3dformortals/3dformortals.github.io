• Elvis operator ?:
    • Used to return the not null value even the conditional expression is null.
    Also used to check the null safety of values. weblink //var str:String = null //error var s:String? = null
    var slen: Int = if (s != null) s.length else -1 //-1
    var elen: Int = s?.length ?: -1 //-1 ...safe s?

    s = "5"
    //println(s.length) //error
    if ( s != null ) println( s.length ) else println( "s == null" ) //1
    if ( s is String ) println("String") //String
    if ( s !is String ) println("not String") else println("String") //String

    var obj:Any? = null
    // println( obj as Any ) //error
    println( obj as Any? ) //null
    obj = 123
    println( obj as Any ) //123
    println( obj as Any? ) //123

    var magic:Any = "bang"
    var safeString: String? = magic as? String //bang
    var safeInt: Int? = magic as? Int //null

    var x:Any? = null
    println( x )
    //println( x!! ) //error , !! == must be not null or arise exception
    var y:Any? = 0
    println( y ) //0
    println( y!! ) //0
