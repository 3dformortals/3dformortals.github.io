• Scope of Extensions
    • weblink
    Base file: // file name: example.kt
    package foo
    class Baz { ... }     Define extension directly under package: package foo.bar //foo.bar just a package name
    import foo //class Baz was imported from example.kt
    fun Baz.goo() { ... }     Using extension outside its declaring package uses import : package com.example.usage

    import foo.bar.goo // importing all extensions by name "goo"
    // or
    import foo.bar.* // importing everything from "foo.bar"

    fun usage( baz: Baz ) { baz.goo() }
