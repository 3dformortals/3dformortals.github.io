• Set of unique elements kotlin.collections
    • Set is collection of unique read-only elements. weblink
    MutableSet is collection of unique read/write elements. weblink var setA:Set<Any> = setOf(1,2,"abc",'x') //[1, 2, abc, x]
    var set = setOf(1,2,"abc",'x') //[1, 2, abc, x]
    println( set.size ) //4
    println( set.elementAt(2) ) //abc
    println( set.indexOf( "abc" ) ) //2
    println( set.first() ) //1
    println( set.last() ) //x
    set = set.plus(3) //[1, 2, abc, x, 3]
    set = set.plus( arrayOf( 4, 5, 6 ) ) //[1, 2, abc, x, 3, 4, 5, 6]
    set = set.minus( 3 ) //[1, 2, abc, x, 4, 5, 6]
    set = set.minus( arrayOf( 4, 5, 6 ) ) //[1, 2, abc, x]
    var setInt:Set<Int> = setOf( 1, 2, 3 ) //[1, 2, 3]
    set = set.union(setInt) //[1, 2, abc, x, 3]
    println( set.drop( 1 ) ) //[2, abc, x, 3]
    println( set.take( 1 ) ) //[1]
    println( set.zip( setInt ) ) //[(1, 1), (2, 2), (abc, 3)]

    var mSet:MutableSet<Any> = mutableSetOf( 1, 2, 3 ) //[1, 2, 3]
    var mset = mutableSetOf<Any>( 1, 2, 3 ) //[1, 2, 3]
    mset.clear() //[ ]
    mset.add( 4 ) //[4]
    mset.addAll(mSet) //[4, 1, 2, 3]
    mset.remove( 1 ) //[4, 2, 3]
    mset.removeAll(mSet) //[4]
