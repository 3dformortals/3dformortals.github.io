    All data class properties declared in the primary constructor, have next functionality: 1- equals() / hashCode() pair 1- toString() of the form "User(name=John, age=42)" 1- componentN() functions corresponding to the properties in their order of declaration 1- copy() function 1-destructuring declarations 1- primary constructor default and named arguments
    Requirements of data class : 1-the primary constructor needs to have at least one parameter 1-all primary constructor parameters need to be marked as val or var 1-data classes cannot be abstract , open , sealed , inner 1-data classes may extend other classes 1-to exclude a property from the generated implementations, declare it inside the class body 1-excluded property hidden for toString() , equals() , hashCode() , copy() , componentN() functionality data class DataTest(var a:Int = 0, val b:Int = 1){ var c:Int = 2 }
    var test = DataTest()
    println( test.toString() ) //dataTest(a=0, b=1)
    println( test.component1() ) //0
    println( test.component2() ) //1
    // println( test.component3() ) //error
    println( test.c )//2

    data class ReturnTest(var x:Int)
    fun testReturn():ReturnTest{ return ReturnTest( 8 ) }
    var testreturn = testReturn()
    println(testreturn.x) //8

    data class ReturnTestMulti(var x:Int, var y:Int)
    fun testReturnMulti(a:Int, b:Int, c:Int):ReturnTestMulti{ return ReturnTestMulti(a-c, b-c) }
    var ( test1, test2 ) = testReturnMulti( 8, 9, 1 ) //destructuring declarations
    println( "test1=$test1 test2=$test2" ) //test1=7 test2=8
