    println( 1.rangeTo( 3 ).toList() ) //[1, 2, 3]
    println( 1.rangeTo( 3 ).toList() ) //[1, 2, 3]
    println( ( 1 until 3 ).toList() ) //[1, 2]
    println( 1.rangeTo( 3 ).step( 2 ).toList() ) //[1, 3]

    println( 3.downTo( 1 ).toList() ) //[3, 2, 1]
    println( 3.downTo(1).step(2).toList() ) //[3, 1]
    println( 3.downTo(1).step(2).reversed().toList() ) //[1, 3]
    println( 3.downTo(1).reversed().step(2).toList() ) //[1, 3]
    for (i in 3 downTo 1 step 2) print(i) //31
