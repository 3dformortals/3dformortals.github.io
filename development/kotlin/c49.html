    Sorting map by values, without droping elements with similar values, and with only one element from group with unique values(other will droped). val map = mapOf("player1" to 100f, "player2" to 100f, "player3" to 300f, "player4" to 200f)
    /*from min to max values*/
    val smap1 = map.toList().sortedBy{ (_, value) -> value }.toMap()
    val smap2 = map.toList().sortedBy{ it.second }.toMap()
    val smap3 = map.toSortedMap(compareBy{map[it]}) /*in case of similar values only one(first shot) item will in result map, other will droped out*/

    /*from max to min values*/
    val smap4 = map.toList().sortedByDescending{ (_, value) -> value }.toMap()
    val smap5 = map.toList().sortedByDescending{ it.second }.toMap()
    val smap6 = map.toSortedMap(compareByDescending{map[it]}) /*in case of similar values only one(first shot) item will in result map, other will droped out*/

    println("map.keys = ${map.keys}")
    println("sorted from min to max values")
    println("smap1.keys = ${smap1.keys}")
    println("smap2.keys = ${smap2.keys}")
    println("smap3.keys = ${smap3.keys}")
    println("sorted from max to min values")
    println("smap4.keys = ${smap4.keys}")
    println("smap5.keys = ${smap5.keys}")
    println("smap6.keys = ${smap6.keys}")     map.keys = [player1, player2, player3, player4]
    sorted from min to max values
    smap1.keys = [player1, player2, player4, player3]
    smap2.keys = [player1, player2, player4, player3]
    smap3.keys = [player1, player4, player3]
    sorted from max to min values
    smap4.keys = [player3, player4, player1, player2]
    smap5.keys = [player3, player4, player1, player2]
    smap6.keys = [player3, player4, player1]
