    //return defferent types
    fun different_return_type(a:Int,b:Double):Array<Any>{ return arrayOf<Any>( a, b )}
    val (a, b) = different_return_type( 55, 88.8 )
    println( a ) //55
    println( b ) //88.8
    println( a is Int ) //true
    println( b is Double ) //true

    //return nullable different types
    fun nullable_different_return_type(a:Int,b:Double?):Array<Any?>{ return arrayOf<Any?>( a, b )}
    val (c, d) = nullable_different_return_type( 55, null )
    println( c ) //55
    println( d ) //null
    println( c is Int ) //true
    println( d is Double ) //false
