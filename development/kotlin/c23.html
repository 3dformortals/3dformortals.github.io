• Boolean
    • val a = true
    val b = false
    val c:Boolean = false
    val d:Boolean = true

    println( a || b ) //true . one true return true
    println( a || d ) //true . both true return true

    println( a && b ) //false
    println( a && d ) //true
    println( c && b ) //false

    println( !a ) //false
    println( !b ) //true
    println( !c ) //true

    println ( a.and(d) ) //true
    println ( a.and(b) ) //false
    println ( c.and(b) ) //false

    println ( a.compareTo(b) ) //1
    println ( b.compareTo(a) ) //-1
    println ( c.compareTo(b) ) //0

    println ( a.not() ) //false
    println ( b.not() ) //true

    println ( a.or( b ) ) //true
    println ( b.or( c ) ) //false
    println ( a.or( d ) ) //true . both true return true
    println ( d.or( c ) ) //true

    println ( a.xor( b ) ) //true
    println ( b.xor( c ) ) //false
    println ( a.xor( d ) ) //false . only one true return true, both true return false
    println ( d.xor( c ) ) //true     In .and() and .or() case always will be evaluated both parameters.
    An .xor() case always evaluate both parameters.
